All ladder operations hit the **Superman Python** backend (`https://superman.superform.xyz`). The same JWT used for Erebor works here.

<Note>
The Superman Python backend does not yet have an OpenAPI spec. Endpoint shapes documented here are derived from source code analysis.
</Note>

Common automation patterns:
- **APY-based rebalancing**: sort deposit priority by current yield source APY
- **Liquidity-aware redemption**: deprioritize illiquid sources during high redemption demand
- **Dynamic reserve adjustment**: increase reserve ahead of anticipated redemption spikes

## Reading All Ladders

```python
import requests

SUPERMAN_URL = "https://superman.superform.xyz"
CHAIN_ID = 8453
VAULT = "0xYourVault..."
headers = {"Authorization": "Bearer <jwt>"}

resp = requests.get(f"{SUPERMAN_URL}/ladder/ladders", headers=headers)
ladders = resp.json()

vault_ladder = next(
    (l for l in ladders if l["vault_address"].lower() == VAULT.lower()),
    None
)

print("Deposit priority:", vault_ladder["deposit_priority"])
print("Reserve:", vault_ladder["reserve_percent"], "%")
print("Enabled:", vault_ladder["is_enabled"])
```

## Deposit Priority

```python
def update_deposit_priority(chain_id: int, vault: str, ordered_addresses: list[str]):
    """First address = highest priority. All addresses must be whitelisted yield sources."""
    resp = requests.put(
        f"{SUPERMAN_URL}/ladder/{chain_id}/{vault}/deposit-priority",
        headers=headers,
        json={"deposit_priority": [{"address": a} for a in ordered_addresses]}
    )
    resp.raise_for_status()
    return resp.json()
```

## Example: APY-Based Rebalancing

```python
def rebalance_by_apy(chain_id: int, vault: str, apy_data: dict[str, float]):
    """apy_data: { "0xYieldSource1": 0.085, "0xYieldSource2": 0.072, ... }"""
    ladder = get_vault_ladder(chain_id, vault)
    current_order = [item["address"] for item in ladder["deposit_priority"]]

    ranked_order = sorted(current_order, key=lambda addr: apy_data.get(addr, 0), reverse=True)

    if ranked_order != current_order:
        print(f"Rebalancing: {current_order} → {ranked_order}")
        update_deposit_priority(chain_id, vault, ranked_order)
    else:
        print("No rebalance needed")
```

## Example: Dynamic Reserve Adjustment

```python
def adjust_reserve_for_redemption_demand(chain_id: int, vault: str):
    """Increase reserve when redemption queue depth is high."""
    resp = requests.get(
        f"{SUPERMAN_URL}/fulfill-redeems/executions",
        headers=headers,
        params={"chain_id": chain_id, "vault_address": vault, "limit": 100}
    )
    executions = resp.json()["executions"]
    pending = [e for e in executions if e["status"] == "pending"]

    if len(pending) > 5:
        set_reserve(chain_id, vault, 20)
        print(f"High redemption demand ({len(pending)} pending): reserve raised to 20%")
    else:
        set_reserve(chain_id, vault, 5)
        print("Normal conditions: reserve at 5%")
```

## TypeScript Reference

```typescript
const SUPERMAN_URL = "https://superman.superform.xyz";
const headers = { Authorization: `Bearer ${process.env.SUPERMAN_JWT}` };

async function updateDepositPriority(chainId: number, vault: string, addresses: string[]) {
  const resp = await fetch(`${SUPERMAN_URL}/ladder/${chainId}/${vault}/deposit-priority`, {
    method: "PUT",
    headers: { ...headers, "Content-Type": "application/json" },
    body: JSON.stringify({ deposit_priority: addresses.map(a => ({ address: a })) })
  });
  if (!resp.ok) throw new Error(`Update failed: ${resp.status}`);
  return resp.json();
}

async function setReserve(chainId: number, vault: string, percent: number) {
  const resp = await fetch(`${SUPERMAN_URL}/ladder/${chainId}/${vault}/reserve`, {
    method: "PUT",
    headers: { ...headers, "Content-Type": "application/json" },
    body: JSON.stringify({ reserve_percent: percent })
  });
  if (!resp.ok) throw new Error(`Reserve update failed: ${resp.status}`);
}
```
