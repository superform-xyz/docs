Automated service monitoring detects degradation before it impacts depositors and responds faster than manual observation allows.

Key patterns:
- Poll service health every 30s, alert on degradation
- Auto-pause a failing service to prevent cascading issues
- Verify schedule changes propagated to the Celery beat
- Control per-vault services via Erebor for targeted stops

## Monitoring All 7 Services

```python
import requests

SUPERMAN_URL = "https://superman.superform.xyz"
headers = {"Authorization": "Bearer <jwt>"}

def get_service_health() -> list[dict]:
    resp = requests.get(f"{SUPERMAN_URL}/services/status", headers=headers, timeout=10)
    resp.raise_for_status()
    return resp.json().get("services", [])

def get_infra_health() -> list[dict]:
    resp = requests.get(f"{SUPERMAN_URL}/health", headers=headers, timeout=10)
    resp.raise_for_status()
    return resp.json().get("components", [])

def check_all_health():
    services = get_service_health()
    infra = get_infra_health()

    for comp in infra:
        if comp["status"] == "down":
            alert("CRITICAL", f"{comp['component']}: DOWN")
        elif comp["status"] == "degraded":
            alert("WARNING", f"{comp['component']}: degraded")

    critical_services = {"fulfill_deposits", "fulfill_redeems", "tx_manager"}
    for svc in services:
        rate = svc.get("success_rate", 1.0)
        severity = "CRITICAL" if svc["name"] in critical_services else "WARNING"

        if rate < 0.80:
            alert(severity, f"{svc['name']}: {rate:.0%} success rate")
        elif rate < 0.95:
            alert("WARNING", f"{svc['name']}: {rate:.0%} success rate")

        if svc.get("status") == "paused":
            alert("INFO", f"{svc['name']} is currently paused")
```

## Pausing and Resuming Services

```python
def pause_service(service_name: str):
    resp = requests.post(f"{SUPERMAN_URL}/services/{service_name}/pause", headers=headers)
    resp.raise_for_status()
    print(f"Paused: {service_name}")

def resume_service(service_name: str):
    resp = requests.post(f"{SUPERMAN_URL}/services/{service_name}/resume", headers=headers)
    resp.raise_for_status()
    print(f"Resumed: {service_name}")

def auto_pause_if_failing(service_name: str, threshold: float = 0.80):
    services = get_service_health()
    svc = next((s for s in services if s["name"] == service_name), None)
    if svc and svc["success_rate"] < threshold and svc["status"] != "paused":
        alert("CRITICAL", f"Auto-pausing {service_name}: {svc['success_rate']:.0%} success rate")
        pause_service(service_name)
```

## Updating Schedules

```python
import time

def update_schedule_and_verify(service_name: str, interval_seconds: int, timeout: int = 30):
    """Update interval and wait for Celery beat to confirm."""
    resp = requests.put(
        f"{SUPERMAN_URL}/services/schedules/{service_name}",
        headers=headers,
        json={"interval_seconds": interval_seconds}
    )
    resp.raise_for_status()
    print(f"Schedule update submitted for {service_name}: {interval_seconds}s")

    delay = 2
    start = time.time()
    while time.time() - start < timeout:
        status = requests.get(
            f"{SUPERMAN_URL}/services/scheduler/status",
            headers=headers
        ).json()

        if service_name in status.get("loaded_schedules", []):
            print(f"Beat confirmed new schedule for {service_name}")
            return True

        time.sleep(delay)
        delay = min(delay * 2, 10)

    print(f"WARNING: Beat did not confirm schedule update within {timeout}s")
    return False
```

## Per-Vault Service Controls (Erebor)

For vault-scoped control without stopping global workers:

```python
EREBOR_URL = "https://erebor.superform.xyz"

def pause_vault_service(chain_id: int, vault: str, service_name: str, reason: str = ""):
    resp = requests.put(
        f"{EREBOR_URL}/api/v1/vaults/{chain_id}/{vault}/services/{service_name}/pause",
        headers=headers,
        json={"reason": reason}
    )
    resp.raise_for_status()

def resume_vault_service(chain_id: int, vault: str, service_name: str):
    resp = requests.put(
        f"{EREBOR_URL}/api/v1/vaults/{chain_id}/{vault}/services/{service_name}/resume",
        headers=headers
    )
    resp.raise_for_status()

def disable_all_vault_services(chain_id: int, vault: str):
    """Stop all keeper workers for a vault (no gas required)."""
    resp = requests.put(
        f"{EREBOR_URL}/api/v1/vaults/{chain_id}/{vault}/services/enabled",
        headers=headers,
        json={"enabled": False}
    )
    resp.raise_for_status()
    print(f"All services disabled for vault {vault}")

def get_vault_services(chain_id: int, vault: str) -> list[dict]:
    resp = requests.get(
        f"{EREBOR_URL}/api/v1/vaults/{chain_id}/{vault}/services",
        headers=headers
    )
    resp.raise_for_status()
    return resp.json()
```

## Scheduler Health Check

```python
def check_scheduler_health():
    resp = requests.get(f"{SUPERMAN_URL}/services/scheduler/status", headers=headers)
    resp.raise_for_status()
    status = resp.json()

    if status["status"] == "stale":
        alert("WARNING", f"Celery beat is stale — last sync: {status['last_sync_at']}")
    elif status["status"] == "unknown":
        alert("WARNING", "Celery beat status unknown")

    if not status.get("is_leader"):
        alert("INFO", f"Not the beat leader (instance: {status['instance_id']})")

    return status
```

## TypeScript Reference

```typescript
const SUPERMAN_URL = "https://superman.superform.xyz";
const headers = { Authorization: `Bearer ${process.env.SUPERMAN_JWT}` };

async function monitorServices() {
  const [healthResp, servicesResp] = await Promise.all([
    fetch(`${SUPERMAN_URL}/health`, { headers }),
    fetch(`${SUPERMAN_URL}/services/status`, { headers })
  ]);

  const { components } = await healthResp.json();
  const { services } = await servicesResp.json();

  for (const comp of components) {
    if (comp.status === "down") console.error(`[CRITICAL] ${comp.component}: DOWN`);
    else if (comp.status === "degraded") console.warn(`[WARNING] ${comp.component}: degraded`);
  }

  for (const svc of services) {
    const rate = svc.success_rate;
    if (rate < 0.80) console.error(`[CRITICAL] ${svc.name}: ${(rate * 100).toFixed(0)}%`);
    else if (rate < 0.95) console.warn(`[WARNING] ${svc.name}: ${(rate * 100).toFixed(0)}%`);
  }
}

setInterval(monitorServices, 30_000);
```
